public int[] twoSum(int[] nums, int target) {
        if (nums == null || nums.length < 2) {
            return null;
        }
        Map<Integer, Integer> numberArrayMap = new HashMap<Integer, Integer>();
        for (int i = 0; i < nums.length; i++) {
            numberArrayMap.put(nums[i], i);
        }
        for (int i = 0; i < nums.length; i++) {
            int difference = target - nums[i];
            if (numberArrayMap.keySet().contains(difference)&&numberArrayMap.get(difference)!=i) {
                return new int[]{i, numberArrayMap.get(difference)};
            }
        }
        return null;
    }
